import time
import pandas as pd
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Flatten, Dropout
from tensorflow.keras.optimizers import Adam

# --- Benchmark configurations ---
layer_options = [2, 5]
node_options = [10, 20]

results = []

for layers in layer_options:
    for nodes in node_options:
        print(f"\nTraining model with {layers} layers and {nodes} nodes per layer...")
        
        model = Sequential()
        model.add(Dense(nodes, activation='relu', input_shape=(784,)))
        
        # add hidden layers
        for _ in range(layers - 1):
            model.add(Dense(nodes, activation='relu'))
            
        model.add(Dense(10, activation='softmax'))
        model.compile(optimizer=Adam(learning_rate=0.001),
                      loss='categorical_crossentropy',
                      metrics=['accuracy'])
        
        start = time.time()
        history = model.fit(X_train, y_train_cat, epochs=10, batch_size=128,
                            validation_data=(X_val, y_val_cat), verbose=0)
        end = time.time()
        
        train_acc = history.history['accuracy'][-1]
        val_acc = history.history['val_accuracy'][-1]
        
        results.append({
            'Layers': layers,
            'Nodes': nodes,
            'Time (s)': round(end - start, 2),
            'Training Accuracy': round(train_acc, 3),
            'Validation Accuracy': round(val_acc, 3)
        })

# --- Summary table ---
df_results = pd.DataFrame(results)
print("\nBenchmark Results:")
display(df_results)
